#include<list>//类似双向链表，随即存取效率差，不支持[]符号，但是插入和删除效率高
#include<deque>//双端队列，兼顾随即存取，和两端数据的插入删除

using namespace std;
int main()
{
	vector<vector<int>> grid{
		{1,-4,10},
		{3,-2,-1},
		{2,-1,0},
		{0,5,-2},
	};
	int row = grid.size();
	int col = grid[0].size();
	vector<vector<int>> dp1(row,vector<int>(col,0));
	vector<vector<int>> dp2(row, vector<int>(col, 0));
	for (int i = 0; i < row; i++)
		dp1[i][0] = grid[i][0];
	for (int j = 1; j < col; j++) {
		for (int i = 0; i < row; i++) {
			if (i == 0)
				dp1[i][j] = max(dp1[i][j - 1], dp1[i + 1][j - 1]) + grid[i][j];
			else if (i == row-1)
				dp1[i][j] = max(dp1[i][j - 1], dp1[i - 1][j - 1]) + grid[i][j];
			else 
				dp1[i][j] = max(dp1[i][j - 1], max(dp1[i - 1][j - 1],dp1[i+1][j-1])) + grid[i][j];
			
		}
	}
	for (int i = 0; i < row; i++)
		dp2[i][0] = max(grid[i][0], -grid[i][0]);
	for (int j = 1; j < col; j++) {
		for (int i = 0; i < row; i++) {
			if (i == 0)
				dp2[i][j] = max(max(dp1[i][j - 1], dp1[i + 1][j - 1]) - grid[i][j], max(dp2[i][j - 1], dp2[i + 1][j - 1]) + grid[i][j]);
			else if (i == row - 1)
				dp2[i][j] = max(max(dp1[i][j - 1], dp1[i - 1][j - 1]) - grid[i][j], max(dp2[i][j - 1], dp2[i - 1][j - 1]) + grid[i][j]);
			else
				dp2[i][j] = max(max(dp1[i][j - 1], max(dp1[i - 1][j - 1],dp1[i+1][j-1])) - grid[i][j], max(dp2[i][j - 1], max(dp2[i - 1][j - 1], dp2[i + 1][j - 1])) + grid[i][j]);

		}
	}
	int maxval = 0;
	for (int i = 0; i < row; i++) {
		maxval = max(maxval, dp2[i][col - 1]);
	}
	return 0;
}
